/// Checksum returns the 2-byte checksum for the provided data, in little endian
/// byte-order.
pub fn checksum(data: &[u8]) -> [u8; 2] {
    let mut crc: u16 = 0;
    for b in data.iter() {
        crc = (crc << 8) ^ CRC16_TABLE[((crc >> 8) as u8 ^ *b) as usize];
    }
    [(crc & 0xff) as u8, (crc >> 8) as u8]
}

#[cfg(test)]
mod tests {
    use super::checksum;
    extern crate proptest;
    use proptest::prelude::*;

    #[test]
    fn test_checksum() {
        let cs = checksum(b"123456789");
        assert_eq!(cs, [0xc3, 0x31]);
        let cs = checksum(&[0x12, 0x34, 0x56, 0x78, 0x90]);
        assert_eq!(cs, [0xe6, 0x48]);
    }

    proptest! {
        #[test]
        fn test_doesnt_panic(data: Vec<u8>) {
            checksum(&data[..]);
        }
    }
}
